## IsWow64Process2

```c++
BOOL WINAPI IsOS(
  DWORD dwOS
);

void WINAPI GetSystemInfo(
  LPSYSTEM_INFO lpSystemInfo
);

void WINAPI GetNativeSystemInfo(
  LPSYSTEM_INFO lpSystemInfo
);

BOOL WINAPI IsWow64Process(
  HANDLE hProcess,
  PBOOL  Wow64Process
);

BOOL WINAPI IsWow64Process2(
  HANDLE hProcess,
  USHORT *pProcessMachine,
  USHORT *pNativeMachine
);
```
